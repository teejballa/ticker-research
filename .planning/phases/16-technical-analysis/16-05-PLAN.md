### scripts/backfill-technical.ts

```ts
import { config as loadDotenv } from 'dotenv';
loadDotenv({ path: '.env.local' });
import { PrismaClient } from '@prisma/client';
import { PrismaNeon } from '@prisma/adapter-neon';
import { computeTechnicalSnapshot } from '../src/lib/data/technical';

const DRY_RUN = process.argv.includes('--dry-run');
const adapter = new PrismaNeon({ connectionString: process.env.DATABASE_URL! });
const prisma = new PrismaClient({ adapter });

async function main() {
  console.log(`backfill-technical — DRY_RUN=${DRY_RUN}`);

  // Step 1: technical_data on existing snapshots (sequential, 1s throttle).
  const snaps = await prisma.sentimentSnapshot.findMany({
    where: { technical_data: { equals: null as any } },
    orderBy: { scanned_at: 'asc' },
  });
  const histogram: Record<string, number> = {};
  for (const snap of snaps) {
    try {
      const tech = await computeTechnicalSnapshot(snap.ticker, snap.scanned_at);
      const key = tech?.tech_pattern ?? 'null';
      histogram[key] = (histogram[key] ?? 0) + 1;
      if (!DRY_RUN) {
        await prisma.sentimentSnapshot.update({
          where: { id: snap.id },
          data: { technical_data: (tech as object) ?? undefined },
        });
      }
      console.log(`  ${DRY_RUN ? '·' : '✓'} ${snap.ticker} ${snap.scanned_at.toISOString().slice(0,10)} → ${tech?.tech_pattern ?? 'null'}`);
    } catch (err) {
      console.error(`  ✗ ${snap.ticker}: ${(err as Error).message}`);
    }
    await new Promise(r => setTimeout(r, 1000));
  }
  console.log('\nTechPattern distribution:');
  for (const [k, v] of Object.entries(histogram)) console.log(`  ${k}: ${v}`);

  // Step 2: new horizons (30/60/90) on snapshots/reports past those thresholds.
  const now = Date.now();
  const oldEnough = await prisma.sentimentSnapshot.findMany({
    where: { scanned_at: { lte: new Date(now - 30 * 86400_000) } },
    include: { outcomes: true },
    orderBy: { scanned_at: 'asc' },
  });
  for (const snap of oldEnough) {
    for (const day of [30, 60, 90] as const) {
      const target = new Date(snap.scanned_at.getTime() + day * 86400_000);
      if (target.getTime() > now) continue;
      if (snap.outcomes.some(o => o.days_after === day)) continue;
      // fetchHistoricalPrice helper — reuse the same yf.chart() pattern as price-followup
      const price = await fetchHistoricalPrice(snap.ticker, target);
      if (price == null) continue;
      const pct = ((price - (snap.price_at_scan ?? 0)) / (snap.price_at_scan ?? 1)) * 100;
      if (!DRY_RUN) {
        await prisma.priceOutcome.create({
          data: {
            snapshot_id: snap.id,
            days_after: day,
            price,
            pct_change: pct,
            recorded_at: new Date(),
          },
        });
      }
      console.log(`  ${DRY_RUN ? '·' : '✓'} ${snap.ticker} +${day}d → ${pct.toFixed(2)}%`);
      await new Promise(r => setTimeout(r, 500));
    }
  }

  // Same loop for reports.
  // ... same shape as snapshots but reads `report.priceOutcomes` and creates with `report_id` set ...

  console.log('\nDone. Now manually trigger /api/cron/learn with $CRON_SECRET so the recompute pass runs over 288 cells.');
  await prisma.$disconnect();
}

main().catch(err => { console.error(err); process.exit(1); });

import { config as loadDotenv } from 'dotenv';
loadDotenv({ path: '.env.local' });
import { PrismaClient } from '@prisma/client';
import { PrismaNeon } from '@prisma/adapter-neon';

const adapter = new PrismaNeon({ connectionString: process.env.DATABASE_URL! });
const prisma = new PrismaClient({ adapter });

async function main() {
  // Find the cap_class with the most snapshots.
  const counts = await prisma.sentimentSnapshot.groupBy({
    by: ['ticker'], _count: { _all: true },
  });
  // ... aggregate to cap_class via diffusion-trace lookup or community_data.cap_class field ...
  // Simplest: use the LearnedPattern table itself — sum sample_size per cap_class at horizon_days=7.
  const sums = await prisma.learnedPattern.groupBy({
    by: ['cap_class'],
    where: { signal_class: 'technical', horizon_days: 7 },
    _sum: { sample_size: true },
  });
  sums.sort((a, b) => (b._sum.sample_size ?? 0) - (a._sum.sample_size ?? 0));
  const topCap = sums[0]?.cap_class ?? 'large_cap';

  const cells = await prisma.learnedPattern.findMany({
    where: { signal_class: 'technical', cap_class: topCap, horizon_days: 7 },
  });
  const active = cells.filter(c => c.status === 'ACTIVE').length;
  const pct = (active / 8) * 100;  // 8 TechPatterns per cap_class
  console.log(`AC3: ${pct.toFixed(1)}% ACTIVE in cap_class=${topCap} (${active}/8)`);
  await prisma.$disconnect();
  process.exit(pct >= 25 ? 0 : 1);
}

main();
```

### scripts/compare-horizon-brier.ts

```ts
import { config as loadDotenv } from 'dotenv';
loadDotenv({ path: '.env.local' });
import { PrismaClient } from '@prisma/client';
import { PrismaNeon } from '@prisma/adapter-neon';

const adapter = new PrismaNeon({ connectionString: process.env.DATABASE_URL! });
const prisma = new PrismaClient({ adapter });

async function main() {
  const cells = await prisma.learnedPattern.findMany({
    where: { signal_class: 'technical', status: 'ACTIVE', horizon_days: { in: [7, 30] } },
    orderBy: [{ pattern_key: 'asc' }, { horizon_days: 'asc' }],
  });
  // Group by pattern_key, take the AVG brier_in_sample across cap_classes for each horizon.
  const byPattern: Record<string, { '7'?: number; '30'?: number }> = {};
  for (const c of cells) {
    byPattern[c.pattern_key] ??= {};
    const h = c.horizon_days as 7 | 30;
    byPattern[c.pattern_key][String(h) as '7' | '30'] = c.brier_in_sample ?? undefined;
  }
  let improved = 0;
  for (const [pk, b] of Object.entries(byPattern)) {
    if (b['7'] != null && b['30'] != null) {
      const flag = b['30'] <= b['7'] ? '✓' : '✗';
      console.log(`  ${flag} ${pk}: brier_7d=${b['7'].toFixed(3)} brier_30d=${b['30'].toFixed(3)}`);
      if (b['30'] <= b['7']) improved++;
    }
  }
  if (improved > 0) console.log(`AC4: PASS (${improved} pattern(s) improved at 30d vs 7d)`);
  else console.log('AC4: NO_IMPROVEMENT (loose pass — surfacing the truth)');
  await prisma.$disconnect();
  process.exit(0);  // loose pass — AC4 accepts either outcome
}

main();
```

### Step B — Create `/api/insights/horizon-brier/route.ts`

```ts
import { NextResponse } from 'next/server';
import { prisma } from '@/lib/db';
import { adversarialNullBrier } from '@/lib/learning';

export const dynamic = 'force-dynamic';

export async function GET() {
  const cells = await prisma.learnedPattern.findMany({
    where: { signal_class: 'technical' },
    orderBy: [{ pattern_key: 'asc' }, { horizon_days: 'asc' }],
  });
  const TECH_PATTERNS = ['breakout_uptrend','overbought_uptrend','pullback_in_uptrend','consolidation','breakdown','oversold_downtrend','death_cross','golden_cross'] as const;
  const HORIZONS = [3, 7, 14, 30, 60, 90] as const;
  const series = TECH_PATTERNS.map(pk => {
    const points = HORIZONS.map(h => {
      // Aggregate Brier across cap_classes at this (pk, h) — take min sample-size-weighted average.
      const subset = cells.filter(c => c.pattern_key === pk && c.horizon_days === h);
      const totalN = subset.reduce((s, c) => s + c.sample_size, 0);
      const weightedBrier = totalN === 0 ? null
        : subset.reduce((s, c) => s + (c.brier_in_sample ?? 0) * c.sample_size, 0) / totalN;
      const status = subset.some(c => c.status === 'ACTIVE') ? 'ACTIVE'
                   : subset.some(c => c.status === 'EXPLORATORY') ? 'EXPLORATORY'
                   : 'NO_DATA';
      return { horizon_days: h, brier_in_sample: weightedBrier, status };
    });
    return { pattern_key: pk, points };
  });
  return NextResponse.json({ series, brier_null: 0.25 });   // brier_null = 0.25 for binary 50/50 baseline
}
```
