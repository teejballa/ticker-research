3. Add a helper to extract SupplementarySource from settled result (add after the existing settle() helper):
```typescript
function settleSupplementary(
  result: PromiseSettledResult<SupplementarySource | null>,
  sourceName: string,
): SupplementarySource {
  if (result.status === 'fulfilled' && result.value !== null) {
    return result.value;
  }
  // Unexpected error (not a missing key — fetchAlphaVantage/etc handle missing keys internally)
  if (result.status === 'rejected') {
    collection_errors.push(
      `${sourceName}: ${result.reason instanceof Error ? result.reason.message : String(result.reason)}`
    );
  }
  return { name: sourceName, fetched_at: new Date().toISOString(), text_block: '', available: false };
}
```

NOTE: collection_errors must be declared before calling settleSupplementary. The existing `const collection_errors: string[] = [];` is already declared before settle() is called, so settleSupplementary follows the same pattern — declare the helper after collection_errors is initialized.
