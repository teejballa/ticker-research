// src/lib/data/alpha-vantage.ts
// Alpha Vantage OVERVIEW endpoint — supplementary market data source.
// Returns null if ALPHA_VANTAGE_API_KEY env var is missing or request fails.
// Free tier: 25 calls/day at alphavantage.co

import type { SupplementarySource } from '@/lib/types';

const BASE_URL = 'https://www.alphavantage.co/query';

export async function fetchAlphaVantage(ticker: string): Promise<SupplementarySource | null> {
  const apiKey = process.env.ALPHA_VANTAGE_API_KEY;
  if (!apiKey) {
    return {
      name: 'Alpha Vantage',
      fetched_at: new Date().toISOString(),
      text_block: '',
      available: false,
    };
  }

  try {
    const url = `${BASE_URL}?function=OVERVIEW&symbol=${encodeURIComponent(ticker)}&apikey=${apiKey}`;
    const res = await fetch(url, { signal: AbortSignal.timeout(5000) });
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    const data = await res.json() as Record<string, unknown>;

    // Alpha Vantage returns {"Note":"..."} or {"Information":"..."} on rate limit
    if (data.Note || data.Information || !data.Symbol) {
      return { name: 'Alpha Vantage', fetched_at: new Date().toISOString(), text_block: '', available: false };
    }

    const fetched_at = new Date().toISOString();
    const text_block = [
      `=== MARKET DATA: ALPHA VANTAGE ===`,
      `Ticker: ${data.Symbol ?? ticker}`,
      `Company: ${data.Name ?? 'N/A'}`,
      `Sector: ${data.Sector ?? 'N/A'}`,
      `Industry: ${data.Industry ?? 'N/A'}`,
      `Market Cap: ${data.MarketCapitalization ?? 'N/A'}`,
      `P/E Ratio: ${data.PERatio ?? 'N/A'}`,
      `EPS: ${data.EPS ?? 'N/A'}`,
      `Revenue (TTM): ${data.RevenueTTM ?? 'N/A'}`,
      `EBITDA: ${data.EBITDA ?? 'N/A'}`,
      `Gross Profit (TTM): ${data.GrossProfitTTM ?? 'N/A'}`,
      `Profit Margin: ${data.ProfitMargin ?? 'N/A'}`,
      `52-Week High: ${data['52WeekHigh'] ?? 'N/A'}`,
      `52-Week Low: ${data['52WeekLow'] ?? 'N/A'}`,
      `Analyst Target Price: ${data.AnalystTargetPrice ?? 'N/A'}`,
      `Price/Book: ${data.PriceToBookRatio ?? 'N/A'}`,
      `Price/Sales (TTM): ${data.PriceToSalesRatioTTM ?? 'N/A'}`,
      `Data As Of: ${fetched_at}`,
    ].join('\n');

    return { name: 'Alpha Vantage', fetched_at, text_block, available: true };
  } catch {
    return { name: 'Alpha Vantage', fetched_at: new Date().toISOString(), text_block: '', available: false };
  }
}

// src/lib/data/finnhub.ts
// Finnhub company profile + financial metrics — supplementary market data source.
// Returns null if FINNHUB_API_KEY env var is missing or request fails.
// Free tier: 60 calls/min at finnhub.io

import type { SupplementarySource } from '@/lib/types';

const BASE_URL = 'https://finnhub.io/api/v1';

export async function fetchFinnhub(ticker: string): Promise<SupplementarySource | null> {
  const apiKey = process.env.FINNHUB_API_KEY;
  if (!apiKey) {
    return {
      name: 'Finnhub',
      fetched_at: new Date().toISOString(),
      text_block: '',
      available: false,
    };
  }

  try {
    const [profileRes, metricRes] = await Promise.all([
      fetch(`${BASE_URL}/stock/profile2?symbol=${encodeURIComponent(ticker)}&token=${apiKey}`, { signal: AbortSignal.timeout(5000) }),
      fetch(`${BASE_URL}/stock/metric?symbol=${encodeURIComponent(ticker)}&metric=all&token=${apiKey}`, { signal: AbortSignal.timeout(5000) }),
    ]);

    if (!profileRes.ok || !metricRes.ok) throw new Error(`HTTP ${profileRes.status}/${metricRes.status}`);

    const profile = await profileRes.json() as Record<string, unknown>;
    const metricData = await metricRes.json() as { metric?: Record<string, unknown> };
    const metric = metricData.metric ?? {};

    // Finnhub returns empty object {} if ticker not found
    if (!profile.ticker && !profile.name) {
      return { name: 'Finnhub', fetched_at: new Date().toISOString(), text_block: '', available: false };
    }

    const fetched_at = new Date().toISOString();
    // marketCapitalization from Finnhub is in millions — convert to full number for display
    const mktCapRaw = typeof profile.marketCapitalization === 'number' ? profile.marketCapitalization : null;
    const mktCap = mktCapRaw != null ? `$${(mktCapRaw / 1000).toFixed(2)}B` : 'N/A';

    const text_block = [
      `=== MARKET DATA: FINNHUB ===`,
      `Ticker: ${profile.ticker ?? ticker}`,
      `Company: ${profile.name ?? 'N/A'}`,
      `Exchange: ${profile.exchange ?? 'N/A'}`,
      `Country: ${profile.country ?? 'N/A'}`,
      `Market Cap: ${mktCap}`,
      `Shares Outstanding: ${profile.shareOutstanding ?? 'N/A'}`,
      `IPO Date: ${profile.ipo ?? 'N/A'}`,
      `P/E (Normalized Annual): ${metric.peNormalizedAnnual ?? 'N/A'}`,
      `EPS (Normalized Annual): ${metric.epsNormalizedAnnual ?? 'N/A'}`,
      `Revenue/Share (Annual): ${metric.revenuePerShareAnnual ?? 'N/A'}`,
      `Book Value/Share (Annual): ${metric.bookValuePerShareAnnual ?? 'N/A'}`,
      `Debt/Equity (Annual): ${metric.debtToEquityAnnual ?? 'N/A'}`,
      `ROA (FY): ${metric.roaRfy ?? 'N/A'}`,
      `ROE (FY): ${metric.roeRfy ?? 'N/A'}`,
      `P/B: ${metric.pb ?? 'N/A'}`,
      `P/S: ${metric.ps ?? 'N/A'}`,
      `Data As Of: ${fetched_at}`,
    ].join('\n');

    return { name: 'Finnhub', fetched_at, text_block, available: true };
  } catch {
    return { name: 'Finnhub', fetched_at: new Date().toISOString(), text_block: '', available: false };
  }
}

// src/lib/data/fmp.ts
// Financial Modeling Prep company profile + key metrics — supplementary market data source.
// Returns null if FMP_API_KEY env var is missing or request fails.
// Free tier: 250 calls/day at financialmodelingprep.com

import type { SupplementarySource } from '@/lib/types';

const BASE_URL = 'https://financialmodelingprep.com/api/v3';

export async function fetchFMP(ticker: string): Promise<SupplementarySource | null> {
  const apiKey = process.env.FMP_API_KEY;
  if (!apiKey) {
    return {
      name: 'FMP',
      fetched_at: new Date().toISOString(),
      text_block: '',
      available: false,
    };
  }

  try {
    const [profileRes, metricsRes] = await Promise.all([
      fetch(`${BASE_URL}/profile/${encodeURIComponent(ticker)}?apikey=${apiKey}`, { signal: AbortSignal.timeout(5000) }),
      fetch(`${BASE_URL}/key-metrics-ttm/${encodeURIComponent(ticker)}?apikey=${apiKey}`, { signal: AbortSignal.timeout(5000) }),
    ]);

    if (!profileRes.ok || !metricsRes.ok) throw new Error(`HTTP ${profileRes.status}/${metricsRes.status}`);

    const profileArr = await profileRes.json() as Record<string, unknown>[];
    const metricsArr = await metricsRes.json() as Record<string, unknown>[];

    if (!Array.isArray(profileArr) || profileArr.length === 0) {
      return { name: 'FMP', fetched_at: new Date().toISOString(), text_block: '', available: false };
    }

    const profile = profileArr[0];
    const metrics = Array.isArray(metricsArr) && metricsArr.length > 0 ? metricsArr[0] : {};

    const fetched_at = new Date().toISOString();
    const mktCap = typeof profile.mktCap === 'number'
      ? (profile.mktCap >= 1e12 ? `$${(profile.mktCap / 1e12).toFixed(2)}T` : `$${(profile.mktCap / 1e9).toFixed(2)}B`)
      : 'N/A';

    const text_block = [
      `=== MARKET DATA: FMP ===`,
      `Ticker: ${profile.symbol ?? ticker}`,
      `Company: ${profile.companyName ?? 'N/A'}`,
      `Sector: ${profile.sector ?? 'N/A'}`,
      `Industry: ${profile.industry ?? 'N/A'}`,
      `Price: ${profile.price != null ? `$${profile.price}` : 'N/A'}`,
      `Market Cap: ${mktCap}`,
      `P/E Ratio: ${profile.pe ?? metrics.peRatioTTM ?? 'N/A'}`,
      `Revenue/Share (TTM): ${metrics.revenuePerShareTTM ?? 'N/A'}`,
      `P/B Ratio (TTM): ${metrics.pbRatioTTM ?? 'N/A'}`,
      `Debt/Equity (TTM): ${metrics.debtToEquityTTM ?? 'N/A'}`,
      `Description: ${typeof profile.description === 'string' ? profile.description.slice(0, 300) : 'N/A'}`,
      `Data As Of: ${fetched_at}`,
    ].join('\n');

    return { name: 'FMP', fetched_at, text_block, available: true };
  } catch {
    return { name: 'FMP', fetched_at: new Date().toISOString(), text_block: '', available: false };
  }
}
