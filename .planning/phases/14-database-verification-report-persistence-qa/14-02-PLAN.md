// DB-QA-08: Full sign-in → analysis → persist → sign-out → sign-in → history → OPEN flow.
// Uses: NextAuth session cookie injection + page.route() SSE mock + real Neon DB for persistence.

import { test, expect, type BrowserContext } from '@playwright/test';
import { encode } from 'next-auth/jwt';

const TEST_USER_EMAIL = 'e2e-test@cipher.test';
const TEST_TICKER = 'TSLA';

// Minimal fixture AnalysisResult — no real Gemini call
const FIXTURE_RESULT = {
  ticker: TEST_TICKER,
  company_name: 'Tesla Inc.',
  analyzed_at: new Date().toISOString(),
  market_sentiment: 'bullish' as const,
  sentiment_reasoning: 'E2E test fixture — DB persistence verification.',
  bullish_signals: [{ signal: 'E2E test signal', source_citation: 'Test fixture' }],
  bearish_signals: [{ signal: 'E2E risk signal', source_citation: 'Test fixture' }],
  assessment: { buy_pct: 60, hold_pct: 30, sell_pct: 10,
    buy_rationale: 'Test buy.', hold_rationale: 'Test hold.', sell_rationale: 'Test sell.' },
  confidence_level: 'High' as const,
  confidence_explanation: 'E2E fixture — confidence explanation.',
  sources_used: [{ name: 'E2E Test Source', key_fact: 'Test key fact for DB persistence QA' }],
  source_warnings: [],
};

async function injectSession(context: BrowserContext): Promise<void> {
  const secret = process.env.NEXTAUTH_SECRET;
  if (!secret) throw new Error('NEXTAUTH_SECRET not set — check .env.local');
  const token = await encode({
    token: { email: TEST_USER_EMAIL, name: 'E2E Test User', sub: TEST_USER_EMAIL },
    secret,
  });
  await context.addCookies([{
    name: 'next-auth.session-token',  // no __Secure- prefix on localhost
    value: token,
    domain: 'localhost',
    path: '/',
    httpOnly: true,
    sameSite: 'Lax',
  }]);
}

test.describe('db-persistence — DB-QA-08', () => {
  test.afterAll(async ({ request }) => {
    // Clean up: delete all rows written by the test user from Neon
    const secret = process.env.TEST_CLEANUP_SECRET;
    if (secret) {
      await request.delete('/api/test/cleanup', {
        headers: { 'x-test-cleanup-secret': secret },
      });
    }
  });

  test('sign in → run analysis → sign out → sign in → history shows report → OPEN loads report page', async ({ browser }) => {
    // ── Phase 1: Run analysis as test user ──────────────────────────────────
    const ctx1 = await browser.newContext();
    const page1 = await ctx1.newPage();

    // Inject NextAuth session — skips real Google OAuth
    await injectSession(ctx1);

    // Mock source package collection (skip real yahoo-finance2 + Anthropic calls)
    await page1.route(`/api/research/${TEST_TICKER}`, async route => {
      await route.fulfill({
        status: 200,
        contentType: 'application/json',
        body: JSON.stringify({
          filePath: `/tmp/e2e-${TEST_TICKER}.json`,
          ticker: TEST_TICKER,
          assembled_at: new Date().toISOString(),
        }),
      });
    });

    // Mock Gemini analysis SSE stream — returns fixture result immediately
    await page1.route(`/api/analysis/${TEST_TICKER}`, async route => {
      const body =
        `data: ${JSON.stringify({ type: 'progress', message: 'Analyzing...' })}\n\n` +
        `data: ${JSON.stringify({ type: 'result', data: FIXTURE_RESULT })}\n\n`;
      await route.fulfill({
        status: 200,
        headers: { 'Content-Type': 'text/event-stream' },
        body,
      });
    });

    // Navigate to the research page which triggers analysis via ?report= or pipeline
    // For this test, navigate directly to trigger the analysis flow
    await page1.goto(`/research/${TEST_TICKER}`);

    // Wait for the fixture result to be processed and written to Neon
    // The analysis route writes to DB after receiving the result event
    // Wait for either the report page to render or the history to update
    await page1.waitForTimeout(3000); // allow SSE + DB write to complete

    await ctx1.close();

    // ── Phase 2: New session — verify history shows the report ───────────────
    const ctx2 = await browser.newContext();
    const page2 = await ctx2.newPage();

    // Re-inject the same user's session (simulates sign-out + sign-in)
    await injectSession(ctx2);

    // Navigate to home page where ReportHistory renders
    await page2.goto('/');

    // Wait for history to load (not in loading state)
    await page2.waitForSelector('[data-testid="history-row"]', { timeout: 15000 });

    // Verify the test ticker appears in history
    const historyRows = page2.locator('[data-testid="history-row"]');
    const rowCount = await historyRows.count();
    expect(rowCount).toBeGreaterThan(0);

    // Find the row for our test ticker
    const tslaRow = page2.locator('[data-testid="history-row"]').filter({ hasText: TEST_TICKER });
    await expect(tslaRow).toBeVisible();

    // Click OPEN on the TSLA row
    const openBtn = tslaRow.locator('[data-testid="history-open-btn"]');
    await openBtn.click();

    // Should navigate to /research/TSLA?report=<uuid> — not a 404
    await page2.waitForURL(/\/research\/TSLA/, { timeout: 10000 });

    // Confirm we are NOT on a 404 page
    await expect(page2.locator('h1, [data-testid="report-content"], main')).toBeVisible({ timeout: 10000 });

    // Confirm report renders content from fixture (source from FIXTURE_RESULT.sources_used[0].name)
    await expect(page2.locator('text=E2E Test Source')).toBeVisible({ timeout: 10000 });

    await ctx2.close();
  });
});
