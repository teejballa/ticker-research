        * Exit codes:
        *   0  thresholds preserve historical distribution within ±10pp on all tiers
        *   2  ≥ one tier deviates by > 10pp — adjust thresholds and re-run
        *   1  DB error / unexpected failure
        */

       import { ENGAGEMENT_TIER_THRESHOLDS } from '@/lib/data/lightweight-community-scan';

       interface Distribution { high: number; medium: number; low: number; total: number; }

       async function computeNewDistribution(windowDays: number): Promise<{ reddit: Distribution; hackernews: Distribution }> {

         // Engagement-tier is NOT stored in SentimentObservation; we recompute it from the
         // SentimentSnapshot.community_aggregated JSONB column (joined by ticker + scan time).

         // Parse community_aggregated.highlights[].engagement_signal counts.
         //   ...query implementation...
       }

       function compareDistributions(baseline: Distribution, candidate: Distribution): { tier: string; baseline_pct: number; new_pct: number; delta_pp: number }[] {
         // For each tier ∈ {high, medium, low}:
         //   baseline_pct = baseline.tier / baseline.total
         //   new_pct = candidate.tier / candidate.total
         //   delta_pp = (new_pct - baseline_pct) * 100
       }

       async function main() {
         const args = new Set(process.argv.slice(2));
         if (args.has('--dry-run')) {
           console.log('[calibrate] would query SentimentObservation WHERE source IN (reddit, hackernews) AND fetched_at > NOW() - INTERVAL 7 days');
           process.exit(0);
         }
         const windowDays = 7; // fixed for 30.1 calibration
         try {
           const newDist = await computeNewDistribution(windowDays);
           const baseline = await loadHistoricalFirecrawlDistribution();
           const reddit_cmp = compareDistributions(baseline, newDist.reddit);
           const hn_cmp = compareDistributions(baseline, newDist.hackernews);

             ...hn_cmp.map(r => Math.abs(r.delta_pp)),

           console.log('# Engagement-Tier Calibration Report');

           console.log('## HackerNews'); console.table(hn_cmp);

             process.exit(2);
           }
           console.log('PASS: thresholds preserve historical distribution within ±10pp.');
           process.exit(0);
         } catch (e) {
           console.error('[calibrate] ERROR:', e instanceof Error ? e.stack : String(e));
           process.exit(1);
         }
       }

        /**

         * 'firecrawl', 'FIRECRAWL', or '@mendable/firecrawl-js'. Exits 1 on any hit.

        import { execSync } from 'child_process';
        import * as path from 'path';

        const PATTERNS = ['firecrawl', 'FIRECRAWL', '@mendable/firecrawl-js'];
        const SCAN_DIRS = ['src', 'tests', 'scripts'];
        const ALLOWLIST = new Set([
          path.normalize('scripts/check-firecrawl-removed.ts'),
          path.normalize('tests/scripts/check-firecrawl-removed.unit.test.ts'),
        ]);

          for (const dir of SCAN_DIRS) {
