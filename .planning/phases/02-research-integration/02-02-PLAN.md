<feature>
  <name>Research Brief Formatter</name>
  <files>src/lib/research-brief.ts, src/lib/__tests__/research-brief.test.ts</files>
  <behavior>
    formatResearchBrief(pkg: SourcePackage): string
    - Returns string containing all 6 section headers (MARKET DATA, FUNDAMENTALS, ANALYST SENTIMENT, SEC FILINGS, SOCIAL SENTIMENT, COLLECTION NOTES)
    - Header line: `=== TICKER RESEARCH BRIEF: {ticker.toUpperCase()} ===`
    - Null values formatted as "N/A"
    - Numbers: price/target formatted with $ prefix and 2 decimal places; market_cap formatted as $X.XT/B/M using human-readable shorthand; revenue same; percentages with + or - prefix and 2 decimal places
    - analyst_sentiment.recent_changes: list each as "  - {analyst} at {firm} ({action}, {date})"
    - sec_filing_summary: "Most Recent 10-K: {text or N/A}" and "Most Recent 10-Q: {text or N/A}"
    - social_sentiment.signals: bullet list, each as "  - {signal}"
    - collection_errors: if non-empty, each on its own line as "Warning: {error}"
    - Result is ~2000-4000 characters for a typical run

    extractNewsUrls(pkg: SourcePackage): string[]
    - Returns pkg.news.items.map(i => i.url).filter(url => url && url.trim() !== '')
    - Deduplicate: filter out URLs already seen (preserve order)
    - Cap at 15 URLs maximum
    - Returns empty array if news.items is empty or undefined
  </behavior>
