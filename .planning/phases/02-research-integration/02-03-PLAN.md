interface AnalysisSignal { signal: string; source_citation: string; }
interface BuySellBreakdown {
  buy_pct: number; hold_pct: number; sell_pct: number;

}

interface AnalysisResult {
  ticker: string; company_name: string; analyzed_at: string;

  confidence_explanation: string;
  sources_used: AnalysisSource[];
  source_warnings: string[];
}
