    ```typescript
    import { getRedis } from '@/lib/data/cache/upstash';
    import type { ProviderId } from '@/lib/telemetry/cost-estimators';

    export class BreakerOpenError extends Error {
      readonly name = 'BreakerOpenError';
      constructor(
        public readonly provider_id: ProviderId,
        public readonly opened_at: number,
      ) {
        super(`Circuit breaker open for ${provider_id} since ${new Date(opened_at).toISOString()}`);
      }
    }

    export interface BreakerConfig {
      ringSize: number;       // 20
      tripErrorRate: number;  // 0.5  (strict >, not >=)
      minRingForTrip: number; // 5    (avoid tripping on first error)
      openMs: number;         // 30_000
    }

    export const DEFAULT_BREAKER_CONFIG: BreakerConfig = {
      ringSize: 20, tripErrorRate: 0.5, minRingForTrip: 5, openMs: 30_000,
    };

    type BreakerState =
      | { status: 'closed' }
      | { status: 'open'; opened_at: number; reason?: string };

    function stateKey(p: ProviderId)  { return `breaker:${p}:state`;  }
    function ringKey(p: ProviderId)   { return `breaker:${p}:ring`;   }
    function probeKey(p: ProviderId)  { return `breaker:${p}:probe`;  }

    async function readState(p: ProviderId): Promise<BreakerState> {
      const r = getRedis(); if (!r) return { status: 'closed' };
      const raw = await r.get<string>(stateKey(p));
      if (!raw) return { status: 'closed' };
      try { return JSON.parse(raw) as BreakerState; } catch { return { status: 'closed' }; }
    }

    async function openBreaker(p: ProviderId, reason?: string): Promise<number> {
      const r = getRedis(); if (!r) return Date.now();
      const opened_at = Date.now();
      await r.set(stateKey(p), JSON.stringify({ status: 'open', opened_at, reason }), { ex: 3600 });
      return opened_at;
    }

    async function closeBreaker(p: ProviderId): Promise<void> {
      const r = getRedis(); if (!r) return;
      await r.del(stateKey(p));
      await r.del(ringKey(p));
    }

    async function recordOutcome(p: ProviderId, outcome: 'ok' | 'err', cfg: BreakerConfig) {
      const r = getRedis(); if (!r) return;
      await r.lpush(ringKey(p), outcome);
      await r.ltrim(ringKey(p), 0, cfg.ringSize - 1);
      const ring = await r.lrange(ringKey(p), 0, cfg.ringSize - 1);
      if (ring.length < cfg.minRingForTrip) return;
      const errs = ring.filter(o => o === 'err').length;
      const rate = errs / ring.length;
      if (rate > cfg.tripErrorRate) await openBreaker(p, 'error_rate');
    }

    async function tryAcquireProbe(p: ProviderId, openMs: number): Promise<boolean> {
      const r = getRedis(); if (!r) return true;
      const got = await r.set(probeKey(p), '1', { ex: Math.ceil(openMs / 1000), nx: true });
      return got === 'OK';
    }

    /**
     * Composition contract (LOAD-BEARING, see CONTEXT.md D-07 + R-1 in research):
     *   withTelemetry(provider, () => withBreaker(provider, () => withRetry(() => fn())))
     * - Outermost withTelemetry captures BREAKER_OPEN rows into ProviderCallLog.
     * - withBreaker short-circuits before withRetry so a tripped breaker doesn't consume retry budget.
     * - BreakerOpenError is NOT retried (no `code`, no `status>=500`).
     */
    export async function withBreaker<T>(
      provider_id: ProviderId,
      fn: () => Promise<T>,
      overrides: Partial<BreakerConfig> = {},
    ): Promise<T> {
      const cfg = { ...DEFAULT_BREAKER_CONFIG, ...overrides };
      const state = await readState(provider_id);
      if (state.status === 'open') {
        const elapsed = Date.now() - state.opened_at;
        if (elapsed < cfg.openMs) {
          throw new BreakerOpenError(provider_id, state.opened_at);
        }
        // Half-open: probe lock
        const won = await tryAcquireProbe(provider_id, cfg.openMs);
        if (!won) throw new BreakerOpenError(provider_id, state.opened_at);
        try {
          const value = await fn();
          await closeBreaker(provider_id);
          return value;
        } catch (err) {
          if (err instanceof BreakerOpenError) throw err;
          await openBreaker(provider_id, 'probe_failed');
          throw err;
        }
      }
      // Closed path
      try {
        const value = await fn();
        await recordOutcome(provider_id, 'ok', cfg);
        return value;
      } catch (err) {
        if (err instanceof BreakerOpenError) throw err;
        await recordOutcome(provider_id, 'err', cfg);
        throw err;
      }
    }
    ```
